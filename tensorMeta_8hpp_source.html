191 static std::vector<int> fetchBroadcastedSize(const std::vector<int>& sz1, const std::vector<int>& sz2) {
192 if (!sz1.size() || !sz2.size()) {
193 throw std::length_error("Tensor should have atleats a dimension!");
194 std::exit(-1);
195 }
196
197 std::vector<int> curr(sz1.rbegin(), sz1.rend()), incm(sz2.rbegin(), sz2.rend());
198
199 int idx = 0;
200 int n = curr.size(), m = incm.size();
201 std::vector<int> finSize;
202 while (idx < n || idx < m) {
203 if (idx < n && idx < m) {
204 if (curr[idx] == 0 || incm[idx] == 0) {
205 throw std::runtime_error("Size mismatch in Broadcasting");
206 std::exit(-1);
207 } else if (curr[idx] == incm[idx]) {
208 finSize.push_back(curr[idx]);
209 } else if (curr[idx] == 1) {
210 finSize.push_back(incm[idx]);
211 } else if (incm[idx] == 1) {
212 finSize.push_back(curr[idx]);
213 } else {
214 throw std::length_error("Size mismatch in Broadcasting");
215 }
216 } else if (idx < n) {
217 finSize.push_back(curr[idx]);
218 } else if (idx < m) {
219 finSize.push_back(incm[idx]);
220 }
221 idx++;
222 }
223 std::reverse(finSize.begin(), finSize.end());
224
225 return finSize;
226 }

278 std::function<double(double, double)> op) {

282
283 std::vector<int> stride1 = fetchStride(dat1);
284 std::vector<int> stride2 = fetchStride(dat2);

289 int idx2 = getIndex(indices, dat2.tensorSize, stride2);

295 indices[dim]++;
296 if (indices[dim] < out.tensorSize[dim]) {
297 break;
298 }
299 indices[dim] = 0;
300 }
301 }

571 static bool validateMatmul(const TensorMeta& dat1, const TensorMeta& dat2) {
572 std::vector<int> v1, v2;
573 int dim1 = dat1.ndim(), dim2 = dat2.ndim();
574 v1 = dat1.tensorSize;
575 v2 = dat2.tensorSize;
576
577 if (dim1 == 1 && dim2 == 1) {
578 return v1[0] == v2[0];
579 } else if (dim1 == 1 && dim2 == 2) {
580 // (K, ) x (K, N)
581 return v1[0] == v2[0];
582 } else if (dim1 == 2 && dim2 == 1) {
583 // (M, K) x (K, )
584 return v1[1] == v2[0];
585 } else if (dim1 == 2 && dim2 == 2) {
586 // (M, K) x (K, N)
587 return v1[1] == v2[0];
588 } else if (dim1 == 1 && dim2 > 2) {
589 // Batched MatMul
590 // (K, ) x (A, B, ..., K, N)
591 return v1[0] == v2[dim2 - 2];
592 } else if (dim1 > 2 && dim2 == 1) {
593 // Batched MatMul
594 // (A, B, ..., M, K) x (K, )
595 return v1[dim1 - 1] == v2[0];
596 } else {
597 if (v1[dim1 - 1] != v2[dim2 - 2])
598 return false;
599 std::vector<int> v1Part(v1.begin(), v1.end() - 2), v2Part(v2.begin(), v2.end() - 2);
600 try {
601 if (!v1Part.size())
602 v1Part = {1};
603 if (!v2Part.size())
604 v2Part = {1};
605 // If broadcastable
606 std::vector<int> bcSize = fetchBroadcastedSize(v1Part, v2Part);
607 return true;
608 } catch (const std::exception& e) {
609 // If not broadcastable
610 std::cerr << e.what() << std::endl;
611 return false;
612 }
613 }
614 }

623 static std::vector<int> fetchMatmulSize(const TensorMeta& dat1, const TensorMeta& dat2) {
624 bool execFlag = validateMatmul(dat1, dat2);
625 if (!execFlag) {
626 throw std::runtime_error("Shape mismatch for MatMul\n");
627 }
628 std::vector<int> sz1(dat1.tensorSize), sz2(dat2.tensorSize);
629 int M = sz1[sz1.size() - 2], K = sz1[sz1.size() - 1], N = sz2[sz2.size() - 1];
630
631 // If both Meta Storage are 2d Matrix
632 if (sz1.size() == 2 && sz2.size() == 2) {
633 return {M, N};
634 }
635
636 // Partition extra dim to fetch broadcasted size of batches
637 std::vector<int> v1Part(sz1.begin(), sz1.end() - 2), v2Part(sz2.begin(), sz2.end() - 2);
638 if (!v1Part.size())
639 v1Part = {1};
640 if (!v2Part.size())
641 v2Part = {1};
642 std::vector<int> matmulSize = fetchBroadcastedSize(v1Part, v2Part);
643
644 // Add M, N dimension where
645 // dat1 -> ..., M, K
646 // dat2 -> ..., K, N
647 matmulSize.push_back(M);
648 matmulSize.push_back(N);
649
650 return matmulSize;
651 }

674 &out[offSetOut], N);

701 static TensorMeta matmulBroadcast(const TensorMeta& dat1, const TensorMeta& dat2) {
702 std::vector<int> outShape = fetchMatmulSize(dat1, dat2);
703 TensorMeta out(outShape);
704
705 int batchSize = 1;
706 for (int dimIdx = 0; dimIdx < out.ndim() - 2; ++dimIdx) {
707 batchSize *= out.tensorSize[dimIdx];
708 }
709
710 int M = outShape[outShape.size() - 2];
711 int N = outShape[outShape.size() - 1];
712 int K = dat1.tensorSize.back();
713
714 std::vector<int> stride1 = fetchStride(dat1);
715 std::vector<int> stride2 = fetchStride(dat2);
716 std::vector<int> strideOut = fetchStride(out);
717
718 std::vector<int> indices(outShape.size() - 2, 0);
719
720 for (int batchIdx = 0; batchIdx < batchSize; ++batchIdx) {
721 int offSet1 = getMatmulBatchIndex(dat1.tensorSize, stride1, indices);
722 int offSet2 = getMatmulBatchIndex(dat2.tensorSize, stride2, indices);
723 int offSetOut = batchIdx * (M * N);
724
725 matmulAtomic(dat1.rawData, dat2.rawData, out.rawData, offSet1, offSet2, offSetOut, M, K, N);
726
727 // Update indices for broadcasting
728 for (int dim = indices.size() - 1; dim >= 0; --dim) {
729 indices[dim]++;
730 if (indices[dim] < outShape[dim]) {
731 break;
732 }
733 indices[dim] = 0;
734 }
735 }
736
737 return out;
738 }

747 static TensorMeta matmul(const TensorMeta& dat1, const TensorMeta& dat2) {
748 bool matmulFlag = validateMatmul(dat1, dat2);
749
750 if (!matmulFlag)
751 throw std::runtime_error("Inconsistent data dimension, unable to perform matmul!");
752
753 int dim1 = dat1.ndim(), dim2 = dat2.ndim();
754
755 if (dim1 == 1 && dim2 == 1) {
756 // Performing Dot Product
757
758 // A -> (M, ) , B -> (M, )
759 // A -> (1, 1, M), B -> (1, M, 1)
760 TensorMeta dat1Brodcasted = dat1.unsqueeze().unsqueeze();
761 TensorMeta dat2Broadcasted = dat2.unsqueeze(1).unsqueeze();
762
763 // Perform MatMul
764 // Out -> (1, 1, 1)
765 TensorMeta out = matmulBroadcast(dat1Brodcasted, dat2Broadcasted);
766
767 // Out -> (1,)
768 out = out.squeeze(2).squeeze();
769 return out;
770 } else if (dim1 == 2 && dim2 == 2) {
771 // Expanding last dimension to perform batched matmul
772 // A -> (M, K), B -> (K, N)
773 // A -> (1, M, K), B -> (1, K, N)
774 // Out -> (1, M, N)
775 TensorMeta out = matmul(dat1.unsqueeze(), dat2.unsqueeze());
776
777 // Out -> (M, N)
778 return out.squeeze();
779 } else if (dim1 == 1 && dim2 == 2) {
780 // A -> (M, ) , B -> (M, K)
781 // A -> (1, M), B -> (M, K)(unchanged)
782 // Out -> (1, K)
783 TensorMeta out = matmul(dat1.unsqueeze(), dat2);
784
785 // Out -> (K,)
786 return out.squeeze();
787 } else if (dim1 == 2 && dim2 == 1) {
788 // A -> (M, K) , B -> (K, )
789 // A -> (M, K)(unchanged), B -> (K, 1)
790 // Out -> (M, 1)
791 TensorMeta out = matmul(dat1, dat2.unsqueeze(1));
792
793 // Out -> (M,)
794 return out.squeeze(1);
795 } else {
796 if (dim1 == 1) {
797 // A -> (M, ) , B -> (..., M, K)
798 // A -> (1, 1, M)
799 TensorMeta dat1Brodcasted = dat1.unsqueeze().unsqueeze();
800 // Out -> (..., 1, K)
801 TensorMeta out = matmulBroadcast(dat1Brodcasted, dat2);
802 // Out -> (..., K)
803 return out.squeeze(out.ndim() - 2);
804 } else if (dim1 == 2) {
805 // A -> (M, K), B -> (..., K, N)
806 // A -> (1, M, K)
807 TensorMeta dat1Brodcasted = dat1.unsqueeze();
808 // Out -> (..., M, N)
809 return matmulBroadcast(dat1Brodcasted, dat2);
810 } else if (dim2 == 1) {
811 // A -> (..., M, K), B -> (K, )
812 // B -> (1, K, 1)
813 TensorMeta dat2Brodcasted = dat2.unsqueeze(1).unsqueeze();
814 // Out -> (..., M, 1)
815 TensorMeta out = matmulBroadcast(dat1, dat2Brodcasted);
816 // Out -> (..., M)
817 return out.squeeze(out.ndim() - 1);
818 } else if (dim2 == 2) {
819 // A -> (..., M, K), B -> (K, N)
820 // B -> (1, K, N)
821 TensorMeta dat2Brodcasted = dat2.unsqueeze();
822 // Out -> (..., M, N)
823 return matmulBroadcast(dat1, dat2Brodcasted);
824 } else {
825 return matmulBroadcast(dat1, dat2);
826 }
827 }

899 static std::pair<std::vector<int>, std::vector<int>> fetchBroadcastedAxes(const TensorMeta& base,
900 const TensorMeta& broadcasted) {
901 std::vector<int> axes;
902 std::vector<int> addedDims;
903 int shift = broadcasted.ndim() - base.ndim();
904 for (int i = 0; i < shift; ++i) addedDims.push_back(i);
905 for (int i = 0; i < base.ndim(); ++i) {
906 if (base.tensorSize[i] != broadcasted.tensorSize[i + shift]) {

932 static std::tuple<int, int, int, std::vector<int>> fetchReduceAxInfo(const TensorMeta& meta, int axis,
933 bool keepDims = false) {

935 int jump = 1;
936 int numBatches = 1;
937 for (int i = 0; i < meta.ndim(); i++) {
938 if (i < axis)
939 numBatches *= meta.tensorSize[i];
940 if (i > axis)
941 jump *= meta.tensorSize[i];
942 }
943 int incrementBatchIdx = jump * meta.tensorSize[axis];
944
945 return std::make_tuple(jump, numBatches, incrementBatchIdx, outShape);

961 static TensorMeta reduce(const TensorMeta& meta, std::vector<int> axis, std::function<double(double, double)> op,
962 bool keepDims = false, double initVal = 0) {
963 // If no axis specified then perform on the whole data

965 axis = arange(0, meta.ndim());
966 }

968 sort(axis.rbegin(), axis.rend());

988 static TensorMeta reduceSingle(const TensorMeta& meta, int ax, std::function<double(double, double)> op,
989 bool keepDims = false, double initVal = 0) {
990 auto [jump, numBatches, incrementBatchIdx, outShape] = fetchReduceAxInfo(meta, ax, keepDims);
991 TensorMeta out(outShape);
992 out.updateAll(initVal);
993
994 for (int idx = 0; idx < meta.numel; ++idx) {
995 int outIdx = ((idx / incrementBatchIdx) * jump) + (idx % jump);
996 out.rawData[outIdx] = op(out.rawData[outIdx], meta.rawData[idx]);
997 }
998
999 return out;
