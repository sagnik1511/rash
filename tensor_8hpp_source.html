54 void backward() {
55 // Return if already calculated the gradients
56 if (gradVisited[id]) {
57 return;
58 }
59
60 // Mark gradients calculated
61 gradVisited[id] = true;
62
63 // Return if no backward function found!
64 if (!(requiresGrad && _backward)) {
65 return;
66 }
67
68 // Perform gradient update
69 _backward(grad);
70 _backward = nullptr;
71
72 // Backtrack to previous linked tensors

75 p->backward();
